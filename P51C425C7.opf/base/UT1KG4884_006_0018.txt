༄༅། །གཡུང་གི་ལོ་རྒྱུས་བཞུགས་སོ།

 ། ཨེ་མ་ཧོ། །རབ་དཀར་ལས་དང་སྨོན་ལམ་བཟང་པོས་འཕངས། །ཀུན་མཁྱེན་རྗེ་ཡིས་མ་ནོར་ལམ་བསྟན་པའི། །ཟབ་ལམ་སྙིང་པོ་འདི་ལ་བསྒྲོད་ཚུལ་ནི། །བླ་མེད་བླ་མ་མཆོག་ལ་རབ་གུས་ཏེ། །བདག་གཞན་ཐར་ལམ་ཆེན་པོར་འདུན་པ་འཕངས། །སྤྱི་མཐུན་གྲུབ་པའི་ལམ་དུ་མངོན་སུམ་ནས། །སྣང་ཞིང་སྲིད་པར་འཛིན་པ་འཁོར་བའི་ཆོས། །མ་སྤངས་གནས་སུ་དག་པ་རུ་ཤན་ཕྱེད། །སྟོང་ཉིད་ཕུ་ཐག་ཆོད་པ་རྩ་བྲལ་གཤིས། །ཇི་བཞིན་མངོན་དུ་གྱུར་པར་རྣལ་དུ་དབབ། རིག་པ་གཅེར་བུ་ལྷིངས་ལ་རྟེན་མེད་དུ། །གཟོ་མེད་ཞེན་བྲལ་གསལ་སྟོང་རང་སོར་བཞག །གང་ཤར་གཟོ་མེད་རང་གྲོལ་ཁྲེགས་ཆོད་ཀྱི། ། དོན་ཁྲིད་མཉམ་ཉིད་རོ་གཅིག་དེ་ལྟར་བསྐྱངས། །རང་རིག་གསལ་སྟོང་མངོན་སུམ་རྗེན་པ་ལ། །ཡེངས་མེད་བློ་བྲལ་ཐོག་རྒལ་དོན་ཁྲིད་བསྐྱངས། །སྐྱེ་མེད་སྟོང་པ་གཤིས་ཀྱི་ཕུ་ཐག་ཆོད། །རང་རིག་གསལ་སྟོང་མངོན་སུམ་ཐིལ་དུ་ཕེབས། །གང་ཤར་སྤྲོས་བྲལ་རྗེན་པའི་རང་ཞལ་བལྟས། །འབྲལ་མེད་ཆོས་ཉིད་རོལ་པར་དྲན་པས་མཆོག །གཟུང་འཛིན་རྩིས་གདབ་ཞེན་ཡུལ་མཐའ་དང་བྲལ། །བར་སྣང་ནང་དུ་ནམ་མཁའ་བསྒོམ་པ་ལ། །ཡང་དག་དྲན་པས་རང་སོར་སོ་མར་བཞག །ཨྠིྀ། ༄༅། །རྡོ་རྗེ་ཐེག་པའི་དགོངས་པ་ཡིས། །རིན་ཆེན་སེམས་ཉིད་མངོན་དུ་བསྟན། །ཡང་དག་ཡེ་ཤེས་བདེ་བའི་མཆོག །ལྷུན་གྲུབ་ངང་དུ་གུས་པས་འདུད། །མཁའ་དབྱིངས་ཕྱོགས་མེད་མཐའ་བྲལ་དུ། །འགྲོ་བ་རང་རྩལ་འགག་མེད་པ། །སྙིང་པོ་རྣམས་ལ་མངོན་གྱུར་ནི། །ཐིག་ལེ་སྤྲོས་མེད་ངང་དུ་རྟོགས། །དེའི་ལུང་བླ་མ་དམ་པ་ལས། །བདག་ཅག་སྤྲང་པོ་རི་པ་ཡིས། ། དབེན་གནས་བྲག་ནག་ཕོ་བྲང་དུ། །ལྕགས་བྱི་ཟླ་བ་གསུམ་པ་སྟེ། །གཏེར་ནས་བྱོན་ལོ་ཞེ་བརྒྱད་ལ། །དབང་ཁྲིད་ལུང་དཔེ་མན་ངག་རྫོགས། །བཏགས་སྒོམ་དྲན་པས་གྲོལ་བའི་ཆོས། །ཡོངས་རྫོགས་ཞུས་ནས་རང་གཞན་དོན། ། གཉིས་ཀ་གྱིས་ཀྱི་རྗེས་གནང་ཐོབ། །གཞན་ཡང་གནས་མཆོག་ཙཱ་རི་ཡི། ། ཡོན་བདག་རྒྱལ་པོ་སྨོན་ལམ་གྱིས། །སྔར་འདས་ལོ་ནི་བཅོ་ལྔའི་གོང༌།

། གཏེར་སྟོན་ཞལ་སློབ་ཤོ་བན་གྱི། །ཕྱག་དཔེ་དད་པའི་ཤུགས་ཀྱིས་ཕུལ། །ཨོ་རྒྱན་རྗེ་དང་གདམས་པ་ལ། །སྔོན་གྱི་ལས་འབྲེལ་བཟང་པོ་ཡི། །ནུས་པ་ཡོད་ངེས་སྙམ་པ་ཡིས། །དགའ་དང་སྤྲོ་བ་ཚད་མེད་སྐྱེས། །དེང་སང་བདག་པོར་ངེས་པ་ན། །དད་བརྩོན་འདུན་པ་ཆེར་བསྐྱེད་དེ། །ཀ་དག་ལྷུན་གྲུབ་འདུས་མ་བྱས། །རང་བཞིན་སྟོང་གསལ་འགག་པ་མེད། །ཀུན་ཁྱབ་དབྱེ་བ་མེད་པའི་དོན། །རྩོལ་བྲལ་ཐེག་མཆོག་རྫོགས་ཆེན་དང༌། ཁྲེགས་ཆོད་ཐོད་རྒལ་ཟུང་འཇུག་དང༌། །བཏགས་གྲོལ་ཐབས་ཆ་ལ་སོགས་པས། །འདི་ཕྱི་འཆི་ཁ་བར་དོ་རུ། །རང་བློ་ནུས་པའི་རིམ་སྟོབས་ཀྱིས། །འཕྱུགས་མེད་སྐུ་གསུམ་རང་ཆས་སུ། །ཐོབ་པ་ལུང་རིག་མྱོང་བ་གསལ། །ཨེ་མ་དེ་ཡང་རྗེ་ཡི་དྲིན། །དྲིན་ལན་བསབ་པའི་མཆོད་པ་འབུལ། །ཕུལ་བའི་འབྲས་བུ་འགྲོ་རྣམས་སེམས། ། སེམས་ཉིད་རྣལ་མ་བློ་བྲལ་དོན། །དོན་དམ་ཇི་བཞིན་རྟོགས་པ་ལ། །ལ་བཟླ་བ་འདི་འདའ་བ་མེད། །ཤེས་བྱའི་གནས་ལུགས་ཇི་བཞིན་སྣང་བ་ལ། །འགག་མེད་རང་གྲོལ་རྟོགས་པ་རྗེ་ཡི་དྲིན། །སྟོན་འཁོར་གཅིག་ཏུ་འདྲེས་པའི་ཡུལ་ལས་འདས། མ་བཅོས་ལྷུན་གྱིས་གྲུབ་པ་གཉུག་མའི་ངང༌། རྩོལ་བྲལ་བྱ་བྱེད་མེད་པ་སོ་མའི་དོན། ཅིས་ཀྱང་ཆོག་པ་འདུག་པས་བློ་རེ་བདེའོ། །དཔལ་ལྡན་བླ་མ་དམ་པ་ཤེས་རབ་ཚུལ་ཁྲིམས་དེ་ཉིད་ལ་དབེན་གནས་བྲག་ནག་ཏུ་དམ་པ་ཆོས་ཉིད་ཤེས་རབ་ཀྱིས་དབང་དང་ཆོས་སྐོར་ཐམས་ཅད་ཡོངས་སུ་རྫོགས་པར་ཐུགས་ཉམས་སུ་བཞེས་ཤིང་ཁྱད་པར་ཁྲིད་ཀྱི་དོམ་ཁྲེགས་ཆོད་ཀྱིས་ཡིན་ལུགས་མངོན་དུ་གྱུར་པར་མཛད་ནས། ཐོད་རྒལ་གྱིས་འཕྱུགས་མེད་གནད་ཀྱིས་མངོན་སུམ་རང་ཐོག་ཏུ་གཏན་ལ་ཕབ་ནས་དུས་དེ་ཉིད་དུ་ཀུན་ཏུ་བཟང་པོའི་དགོངས་པ་མངོན་དུ་མཛད་པའོ། །དམ་པ་པྲ་ཛྙཱ་ཤྲཱི་དེ་ཉིད་ལས་དབེན་གནས་བྲག་ནག་གི་ཕོ་བྲང་དུ་བདག་ནམ་མཁའ་གྲགས་ཀྱིས་དབང་དང་ལུང་རྣམས་ཡོངས་སུ་རྫོགས་པར་ཞུས་ཀྱང༌། ཁྲིད་ཀྱི་ངེས་པ་མ་རྙེད་པ་ལ། དུས་ཕྱིས་སྤྲུལ་སྐུ་རིག་འཛིན་རྒོད་གྱི་ལྡེམ་འཕྲུ་ཅན་དང་མཇལ་དུས་ཆོས་ཀྱི་ཆེ་བ་དང༌།
