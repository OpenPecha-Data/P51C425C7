རྒྱུན་བབ་བསམ་ལ་སྔགས་འདི་བཟླས༔ ཨོཾ་གུ་རུ་དེ་ཝ་ཌཱཀྐི་ནི་ཧ་རི་ནི་ས༔ ཨོཾ་ཨཱ་ཡུ༔ཤེ་སིདྡྷི་ཧཱུྃ༔ ཨོཾ་སུ་པ་ཏེ་ན་ཏེ༔ སུ་པ་ཏེ་ན་ཏེ་ཡེ་སྭཱཧཱ༔ རྫོགས་རིམ་བདག་དང་ལྷ་མོ་གཉིས༔ དབྱེར་མེད་ངང་ལ་བཞག་པར་བྱ༔ དེ་ནས་མཇུག་གི་རིམ་པ་ནི༔ རྩ་བའི་བསྙེན་པ་འབུམ་དང་ཁྲི༔ སྟོང་ལ་སོགས་པ་བཟླས་བྱས་ནས༔ གྲུབ་པའི་རྟགས་ནི་འདི་ལྟར་འབྱུང་༔ ལུས་ལ་གོ་ཆ་གོན་པ་དང་༔ ཆུ་ལས་བརྒལ་བ་རྨི་བ་དང་༔ ཉི་མ་ཤར་དང་གྱེན་ལ་འཛེག༔ འབྲུ་དང་ལོངས་སྤྱོད་རྨི་བ་དང་༔ བདུད་རྩིའི་ཆུ་རྒྱུན་འཐུང་བ་རྨི༔ དེ་ལྟར་རྟགས་རྣམས་བྱུང་བ་དང་༔ བུམ་པ་རང་གི་སྤྱི་གཙུག་བཞག༔ བུམ་པའི་ནང་ནས་སྙིང་དཀྱིལ་བབ༔ སྙིང་པད་འདབ་བཞི་དབུས་དང་ལྔར༔ མི་འགྱུར་རྟག་ཏུ་བཞུགས་པར་བསྒོམ༔ དེ་ནས་བུམ་ཆུ་བཏུང་ཞིང་བཀྲུ༔ བུམ་པའི་མར་རྣམས་རང་ཉིད་བཟའ༔ ལྕེ་ཡི་དཀྱིལ་ཡང་རྐན་ལ་ཟེར་བའང་འདུག་དུ་ཧཱུྃ་བསྒོམས་ཏེ༔ ཀཱ་ཡ་སིདྡྷི་ཨོཾཿཝཱཀྐ་སིདྡྷི་ཨཱཿ ཙིཏྟ་སིདྡྷི་ཧཱུྃ༔ བཟླས་པས་སྐུ་གསུང་ཐུགས་ཐོབ་བསམ༔ འཆི་མེད་ཡེ་ཤེས་བདུད་རྩི་འདི༔ བཞག་པ་ལྕེ་ཡི་དཀྱིལ་དུ་བཞག༔ བབ་པ་ཙིཏྟའི་དཀྱིལ་དུ་བབས༔ དཀར་ཞིང་འཚེར་ལ་སྣུམ་ཞིང་འདྲིལ༔ བདུད་རྩི་འཆི་མེད་དངོས་གྲུབ་མཆོག༔ ཡེ་ཤེས་ལྷ་མོ་ཨ་ལ་ལ་ཧོ༔ ནད་བཞི་བརྒྱ་རྩ་བཞི༔ རྐྱེན་ཉི་ཁྲི་ཆིག་སྟོང་༔ ཡེ་འདྲོག་སུམ་བརྒྱ་དྲུག་ཅུ༔ བགེགས་རིགས་སྟོང་ཕྲག་བརྒྱད་ཅུ༔ བྱང་ཞིང་དག་པར་གྱུར་ཅིག༔ ཅེས་དངོས་གྲུབ་བླང་བར་བྱའོ༔ ཚེ་སྒྲུབ་འདི་ཡི་ཡོན་ཏན་ནི༔ ཚེ་ཟད་པ་ཡི་རྣལ་འབྱོར་ཡང་༔ ཚངས་པའི་ཚེ་དང་ལྡན་པར་འགྱུར༔ ལས་འཕྲོ་ལྡན་ན་ཉི་ཟླའི་ཚེ༔ འཆི་མེད་རྡོ་རྗེ་ལྟ་བུ་འོ༔ འདི་ལ་སྒོམ་སྒྲུབ་སུས་བྱས་པ༔ དུས་མིན་འཆི་བ་བརྒྱད་མི་འབྱུང་༔ གློ་བུར་ནད་དང་གདོན་རྣམས་ཀྱིས༔ དེ་ལ་ཚུགས་པར་མི་འགྱུར་རོ༔ ཚེ་འཕོས་མཁའ་སྤྱོད་གནས་སུ་འགྲོ༔ འཆི་བ་བརྟག་དང་བསླུ་བ་ལ༔ བསླུ་ཐབས་མང་དུ་བཤད་ན་ཡང་༔ ལྷ་མོ་ལྔ་ཡི་ཚེ་སྒྲུབ་འདི༔
