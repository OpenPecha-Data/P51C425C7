༄༅། །འོད་གསལ་སྦྱོང་བའི་གནད་ཡིག་བདུད་རྩི་

རྒྱལ་པོ་བཞུགས་སོ། ། ཆོས་མཆོག་རྒྱ་མཚོ་ཟབ་མོ་ཡི། །མཛོད་བརྙེས་གང་ལ་ཕྱག་འཚལ་ཏེ། ། བླ་མ་རྡོ་རྗེ་སྙིང་པོ་ཡི། །སྤྱོད་མཆོག་བདུད་རྩི་རྒྱལ་པོ་བྲི། །ནགས་ཚལ་མཚོ་གླིང་དུར་ཁྲོད་དང༌། །ལྷ་ཁང་ཕྱུགས་ལྷས་གྲོང་ཁྱེར་དང༌། །ཁང་སྟོང་ཤིང་གཅིག་ཆུ་ངོགས་འགྲམ། །འཇིགས་དང་དགའ་དང་མྱ་ངན་གནས། །སེམས་གཡེང་བྱེད་དེར་རབ་སོང་ལ། །བདེ་བའི་སྟན་ལ་རབ་འདུག་སྟེ། །སྐྱབས་སུ་འགྲོ་དང་སེམས་བསྐྱེད་ནས། །མདུན་གྱི་ནམ་མཁར་བླ་མ་དང༌། །ཡི་དམ་མཁའ་འགྲོ་ཆོས་སྐྱོང་དང༌། །འགྲོ་དྲུག་ལན་ཆགས་གནོད་བྱེད་འདྲེ། །ཐམས་ཅད་བསམ་ལ་རང་གི་ལུས། །ཤ་ཁྲག་འདོད་ཡོན་ལོངས་སྤྱོད་དུ། །གང་ལ་གང་འདོད་ཆགས་མེད་སྦྱིན། །དེ་རྗེས་གང་ཡང་མི་བསམ་པར། །ཆོས་ཉིད་མཉམ་པའི་ངང་དུ་ཉལ། །དེ་ཚེ་རང་གི་རྣམ་རྟོག་དང༌། །འཁྲུལ་སྣང་ལྷ་འདྲེའི་ཆོ་འཕྲུལ་ནི། །དངོས་དང་ཉམས་སྣང་རྨི་ལམ་དུ། །རྣམ་པ་སྣ་ཚོགས་འབྱུང་བར་འགྱུར། །དེ་ཚེ་གནད་ཀྱི་གདམས་ངག་ལ། །རྒྱུ་མཚན་བརྟགས་པས་ཤེས་པ་དང༌། །རང་བཞིན་སྙེམས་བྲལ་གླཽད་པའོ། །དང་པོ་རྒྱུ་མཚན་བརྟག་པ་ནི། ། ཆོ་་འཕྲུལ་གང་སྣང་དཔེ་མཚོན་པ། །སྣང་སྲིད་སྣོད་བཅུད་ཐམས་ཅད་ཀྱི། ། གཟུགས་སུ་སྣང་བ་ངོ་བོས་སྟོང༌། །ཆ་ཤས་མེད་ཅིང་རང་བཞིན་བྲལ། །སྒྱུ་མ་རྨི་ལམ་ཆུ་ཟླ་བཞིན། །མ་སྐྱེས་པས་ན་གནས་པ་མེད། །འགགས་མེད་འགྲོ་འོང་ཀུན་ལས་འདས། །དེ་ལ་འཇིགས་པ་ཅི་ཞིག་ཡིན། །སྒྲ་ནི་གྲགས་པ་ཉིད་ནས་སྟོང༌། །དྲི་རོ་རེག་དང་ཆོས་ཀུན་ཡང༌། །ངོ་བོས་སྟོང་སྟེ་གཟུགས་བརྙན་བཞིན། །མེད་སྣང་སྟོང་པའི་རང་བཞིན་ལ། །འཇིགས་དང་དགའ་བ་ཅི་ཞིག་བྱ། །འཇིགས་པའི་ཡུལ་ལ་བརྟགས་དཔྱད་པས། །རང་བཞིན་མེད་པ་ཇི་ལྟ་བར། །འཇིགས་མཁན་སེམས་ལ་ཚུར་བལྟས་པས། །སེམས་ནི་དུས་གསུམ་གཞི་རྩ་བྲལ། །བརྟག་པས་མ་རྙེད་རྒྱུ་མཚན་བྲལ། ཁ་དོག་དབྱིབས་མེད་ངོས་གཟུང་བྲལ། །གང་ཞིག་གང་གིས་དགའ་འཇིགས་སྐྱེད། །དེ་ལྟར་ཡུལ་སེམས་གཟུང་འཛིན་དུ། །མེད་པ་དེ་ཚེ་ཆོས་རྣམས་ཀུན།

།ནམ་མཁའ་ལྟ་བུར་བདག་མེད་པས། །གཞན་དུ་འཛིན་པ་ངང་གིས་སྟོང༌། །དེ་ཉིད་རང་བྱུང་ཡེ་ཤེས་ཏེ། ། རང་གསལ་སྤྲོས་དང་བྲལ་བར་གཞག །གཉིས་པ་རང་ཞིར་གློད་པ་ནི། བརྟགས་དང་མ་བརྟགས་གང་ཚེ་ཡང༌། །རང་བཞིན་མེད་དེ་གཟུགས་བརྙན་བཞིན། །དེ་ལ་དྲན་བསམ་དགོས་པ་མེད། །གང་སྣང་སྟེང་དུ་གང་ཤར་གློད། ། སྣང་སྲིད་འཁོར་འདས་ཐམས་ཅད་ཀུན། །སྣང་ན་ཆོས་ཉིད་སྟོང་པར་སྣང༌། ། སྟོང་ན་སྲིད་གསུམ་འཁོར་བ་སྟོང༌། །རྟག་ན་ཆོས་སྐུ་འགྱུར་བ་མེད། །ཆད་ན་ལས་དང་ཉོན་མོངས་ཆད། །དེས་ན་གང་སྣང་མི་སྣང་མེད། །ལེགས་ཉེས་ཤོར་གོད་འགའ་མེད་པས། །གཞི་བྲལ་མཉམ་པའི་ངང་དུ་ཉལ། །ལྷ་འདྲེར་སྣང་ཡང་ངོས་གཟུང་མེད། །དེ་ལ་གཟུགས་མེད་འཇིགས་མེད་དགོས། །འཇིགས་མཁན་སེམས་ཡིན་ངོས་གཟུང་མེད། །དེ་ལ་བདེན་མེད་སྤང་མི་དགོས། །དེ་ལྟར་ཡུལ་སེམས་དགའ་འཇིགས་དང༌། །དུག་ལྔའི་རྣམ་རྟོག་ཐམས་ཅད་ལ། ། ཅེར་མཐོང་རང་སངས་ངོས་གཟུང་ལ། །སྙེམས་མེད་ངང་དུ་གློད་པར་བྱ། །དེ་ནས་གཉན་སར་གཉིས་འཛིན་ཞིག །རེ་དོགས་བྲལ་བའི་ཚར་ཚད་ཐོག ། ཞར་ལ་ཐུན་མོངས་དངོས་གྲུབ་ནི། །ལྷ་འདྲེ་དམ་ཐོགས་སྟོང་གྲོགས་བྱེད། ། རྐྱེན་རྣམས་ཐུགས་ཕྲད་རང་གྲོལ་ཡིན། །དེ་ཕྱིར་གནད་འདི་ཉམས་སུ་ ལོངས། །དགེ་བ་དེ་ཡིས་འགྲོ་བ་ཀུན། །སྐྱེ་མེད་ཀློང་དུ་གྲོལ་བར་ཤོག །འོད་གསལ་སྤྱོད་པའི་གནད་ཡིག ། བདུད་རྩི་རྒྱལ་པོ་ཞེས་བྱ་བ། ཐེག་པ་མཆོག་གི་རྣལ་འབྱོར་པ་ཀློང་ཆེན་རབ་འབྱམས་ཀྱིས་བཀོད་པ་རྫོགས་སོ། །དགེའོ། ། དགེའོ། །དགེའོ།། །། མེས་པོའི་ཤུལ་བཞག་ལས། ཀློང་ཆེན་རབ་འབྱམས་ཀྱི་གསུང་འབུམ། ༼༡༠༽འོད་གསལ་སྦྱོང་བའི་གནད་ཡིག་བདུད་རྩི་རྒྱལ་པོ། - PAGE 112 -- PAGE 111 - - PAGE 110 -
