༄༅། །བདུད་རྩི་དྲི་མེད་བཞུགས་སོ། །

ལེགས་པ་ཐམས་ཅད་འབྱུང་བའི་གནས། །བླ་མ་མཆོག་ལ་ཕྱག་འཚལ་ཏེ། །ཏིང་འཛིན་འོད་གསལ་འབྱུང་བའི་མཛོད། །བདུད་རྩི་དྲི་མེད་གསལ་བར་བྱ། །དུས་བཞི་གནད་ཀྱི་གདམས་པ་དང༌། ཁྲེགས་ཆོད་ཐོད་རྒལ་རྣམ་གཉིས་ཀྱི། །གནད་ནི་རྣམ་པ་གསུམ་དག་ལས། །དང་པོ་དུས་བཞིའི་གནད་ལ་གསུམ། །ཕྱི་དང་ནང་དང་གསང་བ་འོ། །དང་པོ་ཕྱི་ཡི་དུས་བཞི་ནི། །དགུན་དཔྱིད་དབྱར་དང་སྟོན་རྣམས་ལ། །ཟླ་བ་དག་ནི་གསུམ་གསུམ་མོ། །དགུན་དུས་ཆུ་སྟེ་གྲང་ཤས་ཆེ། །གནས་དང་སྤྱོད་ལམ་དྲོད་དང་བསྟུན། །དཔྱིད་ནི་བད་ཀན་ས་ཡི་དུས། །ལྕི་ཞིང་འཐིབས་པས་བསང་བ་གཅེས། །དབྱར་མེ་ཚ་བས་བསིལ་བར་བརྟེན། །སྟོན་རླུང་གཡོ་བས་གསལ་དྭངས་ཕྱེད། །གནས་དང་སྤྱོད་ལམ་ཅི་འཕྲོད་བརྟེན། །ནང་གི་དུས་བཞི་རང་ལུས་ཀྱི། །འབྱུང་བཞི་གནད་ཀྱི་རིམ་པ་ལས། །ཉིན་མཚན་ཆ་ནི་བཞི་བཞིར་དབྱེ། །ཉིན་དུས་ཅིར་སྣང་སྒྱུ་མ་དང༌། །འོད་གསལ་ངང་དུ་རྩལ་སྦྱང་ཞིང༌། །མཚན་དུས་ཤེས་པ་ནང་དུ་གཟུང༌། །སྲོད་ལ་ལྟེ་བའི་ཨ་དམར་ལས། །མེ་འབར་སྤྱི་བོའི་ཧཾ་ལ་ རེག །བདུད་རྩི་དཀར་དམར་སྙིང་ཀར་བབས། །ཐིག་ལེ་མཆོག་ལ་སེམས་གཟུང་བས། །ལྷན་ཅིག་སྐྱེས་པའི་ཡེ་ཤེས་འཆར། །གུང་ལ་སེང་གེའི་ཉལ་ཐབས་ཀྱིས། །སྙིང་ནང་ཨ་དཀར་སེམས་གཟུང་ལ། །མི་དམིགས་ངང་དུ་གཉིད་ལོག་བྱ། །ཐོ་རངས་ཨ་དཀར་སྙིང་ནང་ནས། །སྤྱི་གཙུག་འཕར་ཏེ་གཞུ་གང་གི །ནམ་མཁའི་དཀྱིལ་དུ་གནས་པ་ལ། །སེམས་གཟུང་གསལ་སྟོང་ཡེ་ཤེས་འཆར། །གསང་བའི་དུས་བཞི་རྣམ་རྟོག་གི །བཞི་ཆ་གསུམ་བྲལ་ཆོས་སྐུ་ལ། །རང་རིག་ཡེ་ཤེས་མ་ཡེངས་བསྒོམ། །སྔ་ཕྱི་བར་གསུམ་དུས་གསུམ་གྱི། ། རྣམ་རྟོག་འཕྲོ་འདུ་རང་ཆད་དོ། །མི་རྟོག་ཡེ་ཤེས་ཉིད་དང་སྦྱར། །དེ་ལྟར་དུས་བཞིའི་གནད་གོམས་པས། །ཡང་དག་དོན་ལ་ཉེ་བར་འགྱུར། །གཉིས་པ་ཁྲེགས་ཆོད་གནད་ལ་ནི། །ཤར་ནས་རང་གཞག་གནད་དུ་ཆེ། །མཆོད་པས་དུས་གནད་ངེས་པ་ལས། །མི་འགྱུ་རང་ཡལ་སུས་ཤེས་པ། །འདི་ལ་རྟག་མེད་ཆད་པ་མེད། །དྲན་བསམ་བྲལ་བ་སུས་ཤེས་པ།

།འདི་ལ་འགྲོ་མེད་འོང་བ་མེད། །ཅེར་མཐོང་རང་ངོ་སུས་ཤེས་པ། །འདི་ལ་འཁོར་མེད་མྱང་འདས་ མེད། །འདི་ལྟར་སེམས་ཀྱི་དངོས་པོ་གཅེས། །ཞེས་གསུངས་ཤར་ནས་ངོས་ཟིན་པའི། །ངང་དྭངས་སྒོམ་པ་གསང་བའི་གནད། །གསུམ་པ་ཐོད་རྒལ་གནད་ལ་ནི། །འདུག་སྟངས་གཟིགས་སྟངས་རབ་གཅུན་ཏེ། །མ་བྱེད་མ་རྨུགས་མ་འཕྲོས་པར། །ཉིན་མཚན་སྣང་བའི་འོད་གསལ་བསྒོམ། །ཐབས་ལས་ཤེས་རབ་དངོས་སུ་སྐྱེ། །ཉིན་མཚན་བསྲེ་བ་གནད་དུ་ཆེ། །མི་རྟོག་གསལ་དྭངས་ཕྱེད་པ་ནི། །ཆོས་སྐུའི་འོད་གསལ་སྙིང་པོར་བཤད། །དེ་ཉིད་ཉམས་སུ་བླངས་པ་ཡིས། །རང་བཞིན་རྫོགས་པ་ཆེན་པོར་གྲོལ། །དེ་ལྟར་གནད་གསུམ་གདམས་པའི་མཆོག །རྒྱུད་ལུང་མན་ངག་གནད་བསྡུས་ནས། ། བཀོད་པའི་དགེ་བས་འགྲོ་བ་ཀུན། །ལྷུན་གྲུབ་ཀུན་ཏུ་བཟང་པོར་ཤོག །འོད་གསལ་སྒོམ་པའི་གནད་ཡིག ། བདུད་རྩི་དྲི་མེད་ཅེས་བྱ་བ། ཐེག་པ་མཆོག་གི་རྣལ་འབྱོར་པ་ཀློང་ཆེན་རབ་འབྱམས་ཀྱིས་བཀོད་པ་རྫོགས་སོ།། །། མེས་པོའི་ཤུལ་བཞག་ལས། བླ་མ་ཡང་ཏིག བདུད་རྩི་དྲི་མེད། PAGE - 432 -- PAGE 109 - - PAGE 108 -
