༄༅། །མན་ངག་བརྒྱ་བཅུ་དགུའི་ཐེམ་བྱང་བཞུགས་སོ།། 

ལུང་མེད། རྒྱུད་ཀྱི་དགོངས་པ་མ་ཡིག་མན་ངག་སྙིང་པོའི་དོན་བརྒྱ་དང་བཅུ་དགུ་དག་ཏུ་གྲངས་བཅད་པས༔ གསང་བའི་མན་ངག་༡དུམ་དུམ་ཁྲིགས་དང༌། པོ་ཏི་༢སྨུག་ཆུང༌། འཁྲུལ་གཞི་༣གསལ་བའི་མེ་ལོང་དང༌། འབྱུང་བཞིའི་༤གནད་ཡིག་དང༌། རིན་ཆེན་༥གསེར་ཐིག་དང༌། སྣང་བྱེད་༦སྒྲོན་མ་དང༌། འོད་༧གསལ་ལྡེ་མིག་དང༌། ཤེས་བྱ་སྐོར་༨གསུམ་དང༌། གསང་༩བའི་སྡོམ་བྱང་དང༌། སྒྲ་རྟགས་༡༠འཁོར་ལོ་དང་བཅུ། སྐུ་གསུམ་༡༡ལག་ཁྲིད་དང༌། མཁའ་འགྲོ་༡༢པྲ་ཁྲིད་དང༌། བབས་ལུགས་༡༣གསུམ་སྐོར་དང༌། འཁྲུལ་པའི་སྡེབས་༡༤ཡིག་དང༌། གསང་༡༥ཏིག་ཆེན་པོ་དང༌། གསེར་༡༦བྱང་བཀོལ་བ་དང༌། ལམ་ཁྲིད་༡༧ཆེན་པོ་དང༌། ཡེ་ཤེས་༡༨ལག་ཕྲད་དང༌། བཀའ་སྒྲོམ་༡༩ཆེན་པོ་དང༌། བསམ་གཏན་རང་༢༠གཏད་དང་ཉི་ཤུའོ། །གསལ་༢༡བའི་མེ་ལོང་དང༌། གསང་བའི་༢༢སྙིང་ཁོལ་དང༌། སྣང་བ་༢༣བསྡུས་པ་དང༌། གནས་པའི་༢༤བྱུང་ཁུངས་དང༌། ཁྲིགས་ཆོད་༢༥བདུན་པ་དང༌། ཐོད་རྒལ་༢༦དགུ་པ་དང༌། ལུང་བསྟན་༢༧ཆེན་མོ་དང༌། གླང་ཆེན་༢༨དྲུག་པ་དང༌། སྤྱོད་ལམ་༢༩བསྡུས་པ་དང༌། ལྟ་བའི་གསེར་༣༠བཀོད་དང་སུམ་ཅུ། ༣༡པདྨ་ཀློང་ཡངས་དང༌། ལྷའི་༣༢ཕྱག་གསོར་མ་དང༌། གསེར་གྱི་༣༣ཨུཏྤལ་དང༌། དྲན་པའི་༣༤རྒྱུན་གཅོད་དང༌། རིག་༣༥པ་ལམ་སློང་དང༌། འཕྲ་རྒྱུད་༣༦དྲིལ་བ་དང༌། གསལ་བའི་༣༧སྒྲོན་མེ་དང༌། སེང་གེ་རྩལ་༣༨སྤྲུགས་དང༌། ཨ་ཏི་༣༩གནས་དགོངས་དང༌། དགོངས་དོན་༤༠བཞི་པ་དང་བཞི་བཅུ། རྡོ་རྗེ་ལམ་༤༡བསྟན་དང༌། རིན་ཆེན་འོད་༤༢འབར་དང༌། ལྷག་པའི་འཁོར་༤༣ལོ་དང༌། ཁྱུང་ཆེན་༤༤མཁའ་ལྡིང་དང༌། ལྟ་བའི་༤༥དམིགས་སེལ་དང༌། བླ་མའི་༤༦རིམ་པ་དང༌། དྲང་བའི་ཚིག་༤༧རྒྱུད་དང༌། འཁྲུལ་གཞི་༤༨རྩད་གཅོད་དང༌། སེམས་ཀྱི་༤༩དམིགས་དྲུག་དང༌། བསྒོམ་པའི་༥༠འཁོར་ལོ་དང་ལྔ་བཅུ། དབང་གི་༥༡ལྡེ་མིག་དང༌། དབང་གི་༥༢ལས་བྱང་དང༌། སྤྲོས་མེད་༥༣དབང་གི་མན་ངག་དང༌། དབང་གི་༥༤རྩ་བ་དང༌། དབང་གི་༥༥རིག་བྱེད་དང༌། དབང་གི་༥༦ཕྲ་ཏིག་དང༌། འཁོར་ལོ་༥༧དྲུག་པའི་དབང་དང༌།

དམ་ཚིག་༥༨རབ་འབྱེད་དང༌། དབང་བཀོད་༥༩ཆེན་མོ་དང༌། དབང་སྦྱོར་༦༠ཕྲ་ཁྲིད་དང་དྲུག་ཅུ། རིག་པའི་༦༡མེ་ཏོག་དང༌། གསེར་ལུང་༦༢ཆེན་མོ་དང༌། མན་ངག་༦༣རྩ་བ་དང༌། ལུང་ཏིག་༦༤བཅོ་བརྒྱད་དང༌། རིམ་པ་༦༥དྲུག་པ་དང༌། སེམས་ལུང་༦༦བསྒོམ་ཁྲིགས་དང༌། ཡང་༦༧གསང་དྲིལ་བ་དང༌། སྦས་པའི་༦༨ཉི་མ་དང༌། འཕྲ་༦༩བཀོད་བཅུ་པ་དང༌། ལུང་ཡིག་༧༠བསྡུས་པ་དངབདུན་ཅུ། བདུད་རྩི་ཡང་༧༡ཞུན་དང༌། ལུང་གསུམ་༧༢འདུས་པ་དང༌། གཡུ་ཡིག་༧༣ཅན་དང༌། འཕྲ་ཡི་༧༤གཅན་དང༌། གསེར་ཡིག་༧༥ཅན་དང༌། སྙིང་༧༦གཟེར་ཆེན་པོ་དང༌། ཆུང་༧༧ངུ་དང༌། བསམ་༧༨གཏན་ཉལ་བའི་འཁོར་ལོ་དང༌། ༧༩འཕྲུལ་ཡིག་ཅན་དང༌། ཆུང་བ་༨༠དང་བརྒྱད་ཅུའོ། །མཁའ་༨༡འགྲོའི་མཛོད་དང༌། འཕྲོ་བ་༨༢ཆེན་མོ་དང༌། གསང་བའི་༨༣གཟེར་བུ་དང༌། རྩ་གནས་༨༤འཁོར་ལོ་དང༌། བསམ་གཏན་༨༥བབས་ལུགས་དང༌། རིག་པའི་ས་༨༦འབྱེད་དང། སྣང་བ་༨༧རྒྱས་འདེབས་དང༌། ཡེ་ཤེས་༨༨སྦུབས་འཇུག་དང༌། ལོ་རྒྱུས་༨༩ཆེན་མོ་དང་དགུ་བཅུ། གསང་བའི་༩༡ཞལ་ཆེམས་དང༌། ཐིག་པའི་༩༢རུ་ཤན་དང༌། གསང་ ༩༣ བའི་སྒྲ་ཡིག་དང༌། ཡི་གེའི་སྦྱོར་ ༩༤ བཤད་དང༌། སྒྲ་བཞིའི་ ༩༥ མཚམས་སྦྱོར་དང༌། བརྟན་པའི་ ༩༦ གོ་ཆ་དང༌། འགྱུར་ ༩༧ མེད་ཐིག་ལེ་དང༌། དངོས་པོའི་ ༩༨ ཁ་བསྒྱུར་དང༌། འབྲས་བུའི་༩༩དངོས་འབྱེད་དང༌། སྟོང་པའི་༡༠༠མཐའ་འབྱེད་དང་བརྒྱ་ཐམ་པ། ནམ་ ༡༠༡ མཁའ་ལམ་ཁྱེར་དང༌། ཐི་བྱ་༡༠༢ཚང་འཇུག་དང༌། གསལ་བའི་ ༡༠༣ རིན་ཆེན་འོད་དང༌། ཀ་དག་ཆེན་པོའི་ ༡༠༤ དག་ཡིག་དང༌། ལྷུན་གྲུབ ་༡༠༥ ཡི་གེ་དང༌། ཤེས་ ༡༠༦ རབ་འཁོར་ལོ་དང༌། སྨྲ་བའི་ ༡༠༧ མཚོན་ཆེན་དང༌། ཡེ་ཤེས་ ༡༠༨ ལྔ་བསྐོར་དང༌། འོད་ཀྱི་ ༡༠༩ ཚུལ་བཤད་དང༌། འབྱུང་བའི་ ༡༡༠ གནས་འབྱེད་དང་བཅུ། སྒྲོན་མའི་ ༡༡༡ ཁུངས་གཅོད་དང༌། ཕྱག་རྒྱ་ ༡༡༢ བསྡུས་པ་དང༌། བཀོད་པ་ ༡༡༣ ལྔ་པ་དང༌། རྒྱ་མཚོ་ ༡༡༤ ཨར་གཏད་དང༌། རླུང་གི་ ༡༡༥ སྡོམ་བྱང་དང༌། ངོ་སྤྲོད་ ༡༡༦ གསུམ་པ་དང༌། བབས་ལུགས་ ༡༡༧ བཅུ་གཅིག་དང༌། རྩ་ཏིག་ ༡༡༨ ཆེན་པོ་དང༌། རླུང་གི་ཐེམ་

༡༡༩ ཡིག་དང༌། རིག་པའི་ ༡༢༠ གནས་ས་ཉིད་དང༌། མིག་གི་བསམ་ ༡༢༡ གཏན་དང་བཅུ་དགུའོ། །རྩ་བ་ རྩ་བའི་རྒྱུད་གཉིས་ལ་སོགས་པ་རྫོགས་པ་རང་བྱུང་དང་ཡི་གེ་མེད་པ ་བརྒྱད་ཀྱི་ཚུལ་ལས་ནི། །གོ་རིམ་འཇུག་ཅིང་ ངེས་པར་བསྟན་ན་ རྣམ་བཤད་པ། རྩ་རྒྱུད་ནི་ བསྐྱེད་དང ་ བཤད་ བཤད་རྒྱུད་ནི་རང་ཤར་དང་རང་གྲོལ་ དང་ ་ ཡན་ལག་ རིན་ཆེན་སྤུང་པ་སྐུ་གདུང་འབར་བ ་ དང་ །། ལུང་ ཐལ་འགྱུར་བཀྲ་ཤིས་མཛེས་ལྡན་རིག་པ་གསལ་བའི་རྒྱུད་དེ་རྒྱུད་སྡེ་བཞི་ དང་མན་ངག་ མེ་ལོང་གཉིས་ངོ་སྤྲོད་སྤྲས་པ་མུ་ཏིག་འཕྲེང་བ ་ དགོངས ་ པ་ ཀློང་དྲུག་སྒྲོན་མ་འབར་བ་ཉི་ཟླ་ཁ་སྦྱོར་ དང༌། ། འཇུག་ རང་གྲོལ་གཉིས་སེང་གེ་རྩལ་རྫོགས་ནོར་བུ་ཕྲ་བཀོད་ དང་ ནག་མོ་ཁྲོས་ མཁས་པ་མཆོག་ག ིས ་བསྟན་པ ་ འཛིན་པ་ གང་ཟག་གཅིག་གི་རྐྱེན་དུ་བསྟན་པ་ དང་ ། ། སྐྱོང་བ་དང་ རྫོགས་པ་རང་བྱུང་ལ་སོགས་པ་ བསྲུང་བ་དང་བཅོ་བརྒྱད་དུ་བྱུང༌། ཨྠིྀ། རྣལ་འབྱོར་རྣམ་པ་གསུམ་ལ། རྣལ་འབྱོར་ཆེན་པོ་ནི་རང་ཡི་དམ་ལྷའི་སྐུ་གསུང་ཐུགས་རྣལ་མ་ལ། རང་གི་ལུས་ངག་ཡིད་གསུམ་འབྱོར་པས་ན་རྣལ་འབྱོར་རོ། །ཤིན་ཏུ་རྣལ་འབྱོར་ནི། ཆོས་ཀྱི་དབྱིངས་མ་བཅོས་པའི་རྣལ་མ་ལ། རང་རིག་པ་ཡེ་ཤེས་སུ་འབྱོར་བས་ན་རྣལ་འབྱོར་རོ། །རྫོགས་པ་ཆེན་པོའི་རྣལ་འབྱོར་ནི། ཆོས་ཐམས་ཅད་དེ་བཞིན་ཉིད་རྣལ་མ་ལ། རིག་པ་ཡེ་ཤེས་སུ་འབྱོར་པས་ན་རྫོགས་པ་ཆེན་པོའི་རྣལ་འབྱོར་ཞེས་བྱའོ། །དེ་ཡང་བཀའ་ལས་ཀྱང༌། བདེ་བར་གཤེགས་པའི་སྐུ་གསུང་ཐུགས་དང༌། རང་གི་ལུས་ངག་ཡིད་དང་གསུམ་འབྱོར་པས་ན། རྣལ་འབྱོར་ཆེན་པོ་ཞེས་ནི་དེ་ལ་བྱའོ། །ཆོས་དབྱིངས་མ་བཅོས་རྣལ་མ་ལ། །རང་རིག་ཡེ་ཤེས་བློས་འབྱོར་པས། །ཤིན་ཏུ་རྣལ་འབྱོར་ཞེས་ནི་དེ་ལ་བྱའོ། །ཆོས་ཉིད་མ་བཅོས་རྣམ་མ་ལ། །རང་རིག་ཡེ་ཤེས་བློས་འབྱོར་པས། །རྫོགས་ཆེན་རྣལ་འབྱོར་ཞེས་ནི་དེ་ལ་བྱའོ། །སྐུ་འཇམ་གསུང་འཇམ་ཐུགས་འཇམ་འགྲོ་བ་ཡོངས་ཀྱི་དཔལ། ། རྣལ་འབྱོར་རྣམ་པ་གསུམ་ཡང་དོན་ལ་གཅིག་གོ
